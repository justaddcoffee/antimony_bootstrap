### 3.1 Proposed Antimony Rate Law Templates

```antimony
// APP processing (beta-secretase pathway)
J_BACE_cleavage: APP_Brain => sAPPbeta_Brain + CTFbeta_Brain; V_Brain * Vmax_BACE * APP_Brain / (Km_BACE + APP_Brain)

// AB aggregation (nucleation-dependent)
J_AB_oligomerization: 2 AB42_BrainISF => AB42_oligo_BrainISF; V_BrainISF * k_olig * AB42_BrainISF^2
J_AB_fibrillization: AB42_oligo_BrainISF => AB42_fibril_BrainISF; V_BrainISF * k_fib * AB42_oligo_BrainISF

// AB enzymatic clearance
J_NEP_clearance: AB42_BrainISF => ; V_BrainISF * Vmax_NEP * AB42_BrainISF / (Km_NEP + AB42_BrainISF)
J_IDE_clearance: AB42_BrainISF => ; V_BrainISF * Vmax_IDE * AB42_BrainISF / (Km_IDE + AB42_BrainISF)

// Microglia activation
J_microglia_act: Microglia_resting_Brain => Microglia_activated_Brain; V_Brain * k_act * Microglia_resting_Brain * AB42_plaque_Brain
J_microglia_deact: Microglia_activated_Brain => Microglia_resting_Brain; V_Brain * k_deact * Microglia_activated_Brain

// ROS production
J_ROS_prod: => ROS_Brain; V_Brain * (k_ROS_basal + k_ROS_AB * AB42_oligo_BrainISF)
J_ROS_clearance: ROS_Brain => ; V_Brain * k_antioxidant * ROS_Brain * Antioxidant_Brain

// Tau phosphorylation
J_tau_phos: Tau_Brain => pTau_Brain; V_Brain * k_phos * Tau_Brain * GSK3beta_active_Brain
J_tau_dephos: pTau_Brain => Tau_Brain; V_Brain * k_dephos * pTau_Brain * PP2A_Brain

// BBB AB transport
J_AB_efflux: AB42_BrainISF => AB42_Plasma; k_LRP1 * AB42_BrainISF * LRP1_BBB
J_AB_influx: AB42_Plasma => AB42_BrainISF; k_RAGE * AB42_Plasma * RAGE_BBB
```
